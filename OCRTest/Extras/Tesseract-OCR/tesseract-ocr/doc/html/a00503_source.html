66 DIR128::DIR128( //from fcoord
67  const FCOORD fc //vector to quantize
68  ) {

70 
71  low = 0;
72  if (fc.y () == 0) {
73  if (fc.x () >= 0)
74  dir = 0;
75  else
76  dir = MODULUS / 2;
77  return;
78  }
79  high = MODULUS;
80  do {
81  current = (high + low) / 2;
82  if (dirtab[current] * fc >= 0)
83  low = current;
84  else
85  high = current;
86  }
87  while (high - low > 1);
88  dir = low;
89 }
