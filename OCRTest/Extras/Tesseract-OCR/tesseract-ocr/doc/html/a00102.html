*first_col = col_index - 1;

++margin_columns;

*last_col = col_index;
break;

// column.
if (*first_col < 0) {
// It started in between the previous column and the current column.
*first_col = col_index - 1;
}
if (margin_columns == 0)
*first_spanned_col = col_index;
*last_col = col_index;

*last_col = col_index - 1;

*first_col = col_index - 1;

}

*first_col = col_index - 1; // The last in-between.
