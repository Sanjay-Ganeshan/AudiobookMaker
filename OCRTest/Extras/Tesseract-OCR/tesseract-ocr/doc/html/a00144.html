case BSTT_MATH: // Red box.

break;
case BSTT_DIGIT: // cyan box.
pixRenderBoxArb(pix, box, box_width, 0, 255, 255);
break;
case BSTT_ITALIC: // Green box.
pixRenderBoxArb(pix, box, box_width, 0, 255, 0);
break;

pixRenderBoxArb(pix, box, box_width, 0, 255, 0);
break;
